    jQuery คืออะไร

    jQuery คือ JavaScript Library ได้รับการออกแบบเพื่อให้ง่ายต่อHTML DOM traversal tree และการจัดการ event , ภาพเคลื่อนไหว CSS และ Ajax jQuery เป็น free software Open-Source ที่ได้รับใบอนุญาติจาก MIT ในเดือนพฤษภาคม 2019 jQuery ถูกใช้โดย 73% ของเว็บไซต์ยอดนิยม 10 ล้านเว็บไซต์

    อันนี้เป็นภาพเปรียบเทียบเล็ก ๆ น้อย ๆ

  • การรวบคำสั่งหลาย ๆ บรรทัด ให้สั้นลง

    var hiddenBox = document.getElementById("banner-message");
    var button = document.getElementById("button-container");
    button.onclick = function() {
    hiddenBox.style.display = 'none';

การใช้งาน

jQuery จะทำงานใน tag script

  • การหา selector

    $(selector).action() โดยมี selector แบบพื้นฐานดังนี้
    1. $("#id") เป็นการเลือก object ที่มี id
    2. $(".class") การเลือกด้วย classname
    3. $("*") เป็นการเลือก object ทั้งหมด (all selector)
    4. select โดยชื่อ html tag $("a"), $("p"), $("div")
    5. $("selector1, selector2,.... selectorN") เป็นการเลือก object แบบหลายๆอัน (multi selector)
    6. $("selector:first") first match query

    และ action คือ method ต่าง ๆ ที่จะให้ element นั้น ๆ กระทำ

    $("p:first").css("background-color","yellow");

  • jQuery Event Functions

    ไว้ใช้สำหรับ handle event ต่าง ๆ ที่ผู้ใช้จะกระทำต่อหน้าเว็บ เช่น การคลิก ดับเบิลคลิก การวางเมาส์ เป็นต้น โดยจะมี syntax ดังนี้
    1. document ready function
      syntax: $(document).ready(function)

      เหตุการณ์จะเกิดขึ้นเมื่อมีการโหลดเอกสาร html เสร็จเรียบร้อยแล้ว

      ตัวอย่าง
      $(document).ready(function() {
      $("h2").fadeOut(4000);
      });
             สามารถเขียนย่อได้ดังนี้
      $(function() {
      $("h2").fadeOut(4000);
      });
    2. click method
      syntax: $(selector).click(function)

      เหตุการณ์จะเกิดขึ้นเมื่อมีการคลิกอิลิเมนต์ที่เลือก

      ตัวอย่าง
      $(document).ready(function(){
      $("#btn").click(function(){
      $("#Ball").toggle();
      });
      });

      hide/show
    3. dblclick method
      syntax: $(selector).dblclick(function)

      เหตุการณ์จะเกิดขึ้นเมื่อมีการดับเบิลคลิกอิลิเมนต์ที่เลือก

      $(document).ready(function(){
      let ballColor = "red";
      $("#Ball").dblclick(function(){
      if(ballColor == "red"){
      $(this).css("background", "blue");
      ballColor = "blue";
      } else{
      $(this).css("background", "red");
      ballColor = "red";
      }
      });
      });
    4. focus method
      syntax: $(selector).focus(function)

      เหตุการณ์จะเกิดขึ้นเมื่อเมาส์ไปอยู่ในตำแหน่งโฟกัส

      $(document).ready(function(){
      $("input").focus(function(){
      $(this).css("background", "yellow");
      }); $("input").blur(function(){
      $(this).css("background", "white");
      })
      });
    5. mouseover method
      syntax: $(selector).mouseover(function)

      เหตุการณ์จะเกิดขึ้นเมื่อเมาส์อยู่เหนืออิลิเมนต์ที่เลือก

      $(document).ready(function(){
      $("#Ball").mouseover(function(){
      $(this).css("border-radius", "10px");
      }); $("#Ball").mouseout(function(){
      $(this).css("border-radius", "50%");
      });
      });

  • jQuery Effects

    jQuery Effects คือเอฟเฟคต่างๆของ jQuery ใช้สำหรับการเพิ่มลูกเล่นต่าง ๆ ให้กับหน้าเว็บ
    syntax: $(selector).Effect(speed, callback)
    **speed จะกำหนดเป็น slow, normal, fast หรือกำหนดค่าเป็น ms
    **callback คือ function ที่ถูกเรียกใช้งานเมื่อ animated complete
    ใช้เมื่อเราต้องการให้ jQuery ทำคำสั่งแรกจนเสร็จก่อน แล้วจึงค่อยเริ่มทำคำสั่งที่สอง
    Effect คำอธิบาย
    .hide()ซ่อน element
    .show()แสดง element
    .toggle()สลับระหว่างการซ่อนกับการแสดง element
    .slideDown()เลื่อน element จากบนลงล่าง(แสดง)
    .slideUp()เลื่อน element จากล่างขึ้นบน(ซ่อน)
    .slideToggle()สลับระหว่างการเลื่อนลงกับการเลื่อนขึ้น
    .fadeIn()แสดง element จากจางไปเข้ม
    .fadeOut()ซ่อน element จากเข้มไปจาง
    .fadeTo()ปรับความเข้มของ element
    .animate()สร้าง effect เอง
    Animation
    .animate() เอาไว้ใช่สร้าง animation อย่างง่าย ๆ
    systax: $(selector).animate({params, speed, callback})
    **params ใช้กำหนดลักษณะการเคลื่อนไหวของแอนิเมชั่น
    $(document).ready(function(){
    let state = true; setInterval(()=>{
    state?(state=false): (state=true);
    $("#Ball4").animate({
    borderRadius: state?'50%': '10px',
    left: state?'10%': '70%'
    }, 100, ()=>{
    ?($("#Ball4").css("background", "red")):($("#Ball4").css("background", "blue"))
    });
    }, 1000)
    });

  • jQuery DOM

    DOM = Document Object Model

    DOM คือ การมองส่วนต่างๆของหน้าเว็บให้เป็น object ทำให้เราสามารถแยกแยะ และเข้าถึงแต่ล่ะ object ได้ง่ายขึ้น ทั้งนี้ก็เพื่อให้เราสามารถเรียกใช้ object เหล่านั้นได้

    1. Get content
      Methodคำอธิบาย
      .attr()get & set attribute
      .prop()get & set property
      .html()get & set content ในรูปแบบ html tag
      .text()get & set content ในรูปแบบ text
      .val()get & set ค่าของ form
    2. Manipulation Methods
      Methodคำอธิบาย
      .append()แทรก content ต่อท้าย content ที่มีอยู่ของ selector ที่เลือกไว้
      .before()แทรก content ก่อน selector ที่เลือกไว้
      .after()แทรก content หลัง selector ที่เลือกไว้
      .prepend()แทรก content ข้างหน้า content ที่มีอยู่ของ selector ที่เลือกไว้
      .remove()ลบ element ที่ selector เลือกไว้
      .replaceAll()แทนที่ข้อความ หรือ tag ของ selector
      .wrap()ครอบ selector ที่เลือกไว้ด้วย html tag ต่าง ๆ

  • jQuery Traversing

    จะใช้ในการ "ค้นหา" (หรือเลือก) องค์ประกอบ HTML อยู่บนพื้นฐานของความสัมพันธ์กับองค์ประกอบอื่น ๆ เริ่มต้นด้วยหนึ่งตัวเลือกและย้ายผ่านการคัดเลือกว่าจนกว่าจะถึงองค์ประกอบที่คุณต้องการ

    Ancestors

    **บรรพบุรุษ = คลาสที่ครอบ selector อยู่
    Methodคำอธิบาย
    .parent()คืนค่าบรรพบุรุษ 1 ชั้น
    .parents()คืนค่าบรรพบุรุษทั้งหมด
    .parentsUntil()คืนค่าบรรพบุรุษทั้งหมดระหว่าง tag ที่ใส่ไว้ใน parameter

    Descendants

    **ลูก = คลาสที่ถูก selector ครอบอยู่
    Methodคำอธิบาย
    .children()คืนค่าลูก
    .find()คืนค่าelement ที่ตรงกับ parameter ทั้งหมดที่ถูก selector ครอบอยู่รวมลูกของลูกต่อ ๆ ไปด้วย

    Siblings

    **พี่น้อง = element ที่อยู่ชั้นเดียวกัน
    Methodคำอธิบาย
    .siblings()คืนค่า พี่น้อง
    .next()คืนค่า พี่น้องถัดจาก selector
    .nextAll()คืนค่า พี่น้องทุกตัวที่ถัดจาก selector
    .nextUntil()คืนค่า พี่น้องตั้งแต่ element ที่อยู่ถัดจาก selector ถึง parameter ที่กำหนดไว้
    .prev()คืนค่า พี่น้องก่อน selector
    .prevAll()คืนค่า พี่น้องก่อน selector ทุกตัว
    .prevUntil()คืนค่า พี่น้องตั้งแต่ element ที่อยู่ก่อน selector ถึง parameter ที่กำหนดไว้

    Filtering

    Methodคำอธิบาย
    .first()คืนค่า element ตัวแรกที่ตรงกับ selector
    .last()คืนค่า element ตัวสุดท้ายที่ตรงกับ selector
    .eq()คืนค่า element ที่ตรงกับ selector ตัวที่ n (n = index ที่รับค่ามาจาก parameter)
    .filter()คืนค่า element ที่ตรงกับ selector ทุกตัวตัวที่มีเงื่อนไขตรงกับ parameter
    .not()คืนค่า element ที่ตรงกับ selector ยกเว้นตัวที่มีเงื่อนไขตรงกับ parameter ทุกตัว

  • jQuery Ajax

    Ajax = Asynchronous JavaScript and XML

    Ajax คือ เทคนิคการแรกเปลี่ยนข้อมูลกับ server และอัพเดทหน้าเว็บโดยไม่ต้องมีการ reload ทั้งหน้าเว็บ

    MethodSyntaxคำอธิบาย
    $.ajax()$.ajax(url); หรือ $.ajax(url,[options]);request Ajax aysnc
    .get()$(selector).get(url, [data],[callback]);ส่ง GET request ไปที่ server และคึงข้อมูลออกมา
    .post()$(selector).post(url,[data],[callback],[type]);ส่ง POST request ไปที่ server และ submit ข้อมูลลง server
    .load()$(selector).load(url,[data],[callback]);โหลดข้อมูลเข้าสู่ selector
    $.ajax({
    type: 'GET', url: 'send-ajax-data.php', dataType: "JSON", // data type expected from server success: function (data) {
    console.log(data);
    }, error: function(error) {
    console.log('Error: ' + error);
    });
